decl
   int n,arr[50],i,j,dup;
enddecl

begin

  read(n);
  i=0;

  while(i<n) {
    read(arr[i]);
    i = i+1;
  }
  i=0;
  while(i<n) {
    j=i;
    while(j<n) {
      if(arr[i]>arr[j]) then
        dup = arr[i];
        arr[i] = arr[j];
        arr[j] = dup;
      endif;
      j = j + 1;
    }
    i = i+1;
  }
  i=0;
  while(i<n) {
    write(arr[i]);
    i = i+1;
  }

end;
